% Преамбула ПЗ к УИР, оформление по "Методическим указаниям для студентов
% к оформлению пояснительной записки к УИР/НИР" (кафедра 22 "Кибернетика", НИЯУ МИФИ).
%
% Компилировать XeLaTeX'ом (нужен fontspec для кириллического Times-подобного
% шрифта): xelatex main.tex (дважды, чтобы содержание/ссылки посчитались).
%
% Требования из методички, реализованные ниже:
%  - Times New Roman 12пт, интервал 1.5, выравнивание по ширине
%  - поля: слева 30мм, справа 10мм, сверху 20мм, снизу 20мм
%  - отступ первой строки абзаца 1.25см, интервалы до/после абзаца = 0
%  - заголовки: раздел 1-го уровня 14пт полужирный, 2-го уровня 13пт
%    полужирный, 3-го уровня и ниже 12пт полужирный
%  - номера страниц внизу по центру; титульный лист не нумеруется, реферат
%    считается страницей 2
%  - подписи таблиц НАД таблицей (по ширине, без отступа), подписи рисунков
%    ПОД рисунком (по центру)

\usepackage{fontspec}

% PT Serif — свободный кириллический шрифт, метрически близкий к Times New
% Roman, входит в стандартный TeX Live (пакет paratype). Если на вашей
% машине установлен настоящий Times New Roman — замените имя шрифта ниже.
\setmainfont{PT Serif}[
  BoldFont = PT Serif Bold,
  ItalicFont = PT Serif Italic,
  BoldItalicFont = PT Serif Bold Italic
]
\setmonofont{PT Mono}[Scale=0.92]

\usepackage[left=30mm,right=10mm,top=20mm,bottom=20mm]{geometry}

\titlespacing*{\section}{0pt}{0pt}{12pt}
% Подраздел 2-го уровня — 13пт полужирный
\titleformat{\subsection}{\fontsize{13}{15.6}\selectfont\bfseries}{\thesubsection}{0.5em}{}
\titlespacing*{\subsection}{0pt}{10pt}{6pt}
% Подраздел 3-го уровня и ниже — 12пт полужирный
\titleformat{\subsubsection}{\fontsize{12}{14.4}\selectfont\bfseries}{\thesubsubsection}{0.5em}{}
\titlespacing*{\subsubsection}{0pt}{8pt}{4pt}
